% ================================================= % Fake plotting macros % =================================================
\newcommand{\fakefull}[2]{%
  \addplot[#1, points, forget plot] coordinates {(0,#2) (0.2,#2) (0.4,#2) (0.6,#2) (0.8,#2) (1,#2)};
  \addplot[#1, loess] coordinates {(0,#2) (0.4,{#2+0.02}) (0.8,{#2+0.06}) (1,{#2+0.1})};
}

\newcommand{\fakeref}[2]{%
  \addplot[#1, loess] coordinates {(0,#2) (0.4,{#2+0.02}) (0.8,{#2+0.06}) (1,{#2+0.1})};
}
