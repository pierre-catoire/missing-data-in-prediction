% ---------- Fake point cloud ----------
\newcommand{\fakepoints}[2]{%
  \addplot[
    datapoints,
    draw=#1,
    fill=#1
  ]
  coordinates {
    (0.00,#2) (0.05,#2) (0.10,#2) (0.15,#2) (0.20,#2)
    (0.25,#2) (0.30,#2) (0.35,#2) (0.40,#2) (0.45,#2)
    (0.50,#2) (0.55,#2) (0.60,#2) (0.65,#2) (0.70,#2)
    (0.75,#2) (0.80,#2) (0.85,#2) (0.90,#2) (0.95,#2)
    (1.00,#2)
  };
}

% ---------- Fake smooth curve ----------
\newcommand{\fakeline}[2]{%
  \addplot[
    loessline,
    color=#1
  ]
  coordinates {
    (0.0,#2)
    (0.2,{#2 + 0.01})
    (0.4,{#2 + 0.02})
    (0.6,{#2 + 0.04})
    (0.8,{#2 + 0.07})
    (1.0,{#2 + 0.10})
  };
}
