% ---------- Regular method: points + line ----------
\newcommand{\fakefullmethod}[2]{%
  \addplot[datapoints, draw=#1, fill=#1]
  coordinates {
    (0.0,#2) (0.2,#2) (0.4,#2) (0.6,#2) (0.8,#2) (1.0,#2)
  };
  \addplot[loessline, color=#1]
  coordinates {
    (0.0,#2)
    (0.4,{#2+0.02})
    (0.8,{#2+0.06})
    (1.0,{#2+0.1})
  };
}

% ---------- Reference method: dashed line only ----------
\newcommand{\fakerefmethod}[2]{%
  \addplot[
    loessline,
    dashed,
    color=#1
  ]
  coordinates {
    (0.0,#2)
    (0.4,{#2+0.02})
    (0.8,{#2+0.06})
    (1.0,{#2+0.1})
  };
}
